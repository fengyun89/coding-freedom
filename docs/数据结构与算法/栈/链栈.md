# 链栈

栈的链式存储实现。

```js
class LinkedStack {
  #data = {};
  #count = 0;

  // 入栈
  push(element) {
    this.#data[this.#count++] = element;
  }

  // 出栈
  pop() {
    if (this.isEmpty()) {
      return undefined;
    }
    delete this.#data[--this.#count];
    return this.#data[this.#count];
  }

  // 查看栈顶元素
  peek() {
    return this.#data[this.#count - 1];
  }

  // 判断栈是否为空
  isEmpty() {
    return this.#count === 0;
  }

  // 清空栈
  clear() {
    this.#data = {};
    this.#count = 0;
  }

  // 获取栈中元素的个数
  size() {
    return this.#count;
  }

  // 打印栈中的元素
  print() {
    console.log(Object.values(this.#data).toString());
  }
}
```
